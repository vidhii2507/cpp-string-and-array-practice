#include <iostream>
using namespace std;

int main() {
    string str;
    cin >> str;

    int count = 0;

    for(char ch : str) {
        if((ch>='a'&&ch<='z')||(ch>='A'&&ch<='Z')) {
            if(!(ch=='a'||ch=='e'||ch=='i'||ch=='o'||ch=='u'||
                 ch=='A'||ch=='E'||ch=='I'||ch=='O'||ch=='U'))
                count++;
        }
    }

    cout << count;
}
